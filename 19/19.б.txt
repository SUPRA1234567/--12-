#include <stdio.h>

int main() {
    FILE *ff;
    int base;
    
    ff = fopen("sam", "r"); // Відкриття файлу "sam" для читання
    if (ff == NULL) { // Перевірка, чи файл був успішно відкритий
        printf("Помилка відкриття файлу 'sam'\n");
        return 1; // Повертається код помилки
    }
    
    fscanf(ff, "%d", &base); // Зчитування цілого числа з файлу "sam"
    fclose(ff); // Закриття файлу "sam"
    
    ff = fopen("data", "a"); // Відкриття файлу "data" для дописування (додавання в кінець файлу)
    if (ff == NULL) { // Перевірка, чи файл був успішно відкритий
        printf("Помилка відкриття (або створення) файлу 'data'\n");
        return 1; // Повертається код помилки
    }
    
    fprintf(ff, "sam is %d.\n", base); // Запис тексту разом з числом до файлу "data"
    fclose(ff); // Закриття файлу "data"
    
    return 0; // Повертається 0, що означає успішне завершення програми
}
